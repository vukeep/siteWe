```tsx
'use client'

import { useEffect, useRef } from 'react'
import 'cloudinary-video-player/cld-video-player.min.css'

interface Props {
  publicId: string
  cloudName: string
  autoplay?: boolean
}

export function CloudinaryVideoPlayer({ publicId, cloudName, autoplay }: Props) {
  const playerRef = useRef<HTMLVideoElement>(null)

  useEffect(() => {
    if (playerRef.current) {
      const cloudinary = require('cloudinary-video-player')
      
      const player = cloudinary.videoPlayer(playerRef.current, {
        cloudName,
        controls: true,
        muted: autoplay,
        autoplay,
        fluid: true,
        transformation: {
          streaming_profile: 'auto',
          quality: 'auto'
        }
      })

      player.source(publicId)
    }
  }, [publicId, cloudName, autoplay])

  return <video ref={playerRef} className="cld-video-player" />
}
```
